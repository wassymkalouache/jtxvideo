--Utilisateurs dont le nom est Dominique
SELECT * FROM utilisateurs WHERE prenom='Dominique';

--Utilisateurs qui ont renseigné leur promotion
SELECT * FROM utilisateurs WHERE promotion IS NULL;

--Utilisateurs nés dans les années 80
SELECT * FROM utilisateurs WHERE (naissance > '1979-12-31') AND (naissance < '1990-01-01');

--Utilisateurs dont l'email se termine par .fr
SELECT * FROM utilisateurs WHERE (email LIKE '%.fr');

--Utilisateurs nés en mars
SELECT * FROM utilisateurs WHERE (MONTH(naissance) = 3);

--Utilisateurs nés le mois courant
SELECT * FROM utilisateurs WHERE (MONTH(naissance) = MONTH(NOW()));

--Utilisateurs triés par nom et prénom
SELECT * FROM utilisateurs ORDER BY nom,prenom;

--Utilisateurs triés par date d'anniversaire
SELECT * FROM utilisateurs ORDER BY naissance DESC;

--Noms et prénoms classés par ordre alphabétique des noms de famille
SELECT nom,prenom FROM utilisateurs ORDER BY nom ASC;

--Nom et e-mail des Weasley
SELECT prenom,email FROM utilisateurs WHERE nom='Weasley';

--Noms et prénoms triés par anniversaire
SELECT nom,prenom FROM utilisateurs ORDER BY naissance DESC;
